int noOfDivisors(long num)
{
                long i;
                double s;

                int counter=0;
               
                for(i=1;i<sqrt((double)num);i++)
                                if(num%i==0)
                                                counter++;

                counter*=2;
                s=sqrt((double)num);
                if(s*s==num)
                                counter++;
                return counter;
}
